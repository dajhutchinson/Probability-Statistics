\begin{remark}{Normal Distribution}
  Let $X_1,\dots,X_n\iid N(\mu,\sigma^2)$.
\[\begin{array}{rcl}
  \sum\limits_{i=1}^n\frac1{\sigma^2}(X_i-\mu)^2&\sim&\chi^2_n\\
  \sum\limits_{i=1}^n\frac1{\sigma^2}(X_i-\bar{x})^2&\sim&\chi^2_{n-1}
\end{array}\]
\end{remark}
